# Ejemplo: Sistema de aprendizaje de vocabulario
def practicar_vocabulario():
    diccionario = {"casa": "house", "perro": "dog"}
    aciertos = 0
    for español, inglés in diccionario.items():
        respuesta = input(f"¿Cómo se dice '{español}' en inglés?")
        if respuesta.lower() == inglés:
            print("¡Correcto! 🎉")
            aciertos += 1
        else:
            print(f"La respuesta correcta es: {inglés}")
    print(f"Puntuación: {aciertos}/{len(diccionario)}")

# Ejecutar la función
practicar_vocabulario()

Ejemplo de Código:
# Diccionario bilingüe español-inglés
diccionario = {
    "casa": "house",
    "perro": "dog",
    "gato": "cat"
}

# Función para practicar vocabulario
def practicar_vocabulario():
    aciertos = 0
    for palabra in diccionario:
        respuesta = input(f"¿Cómo se dice '{palabra}' en inglés?")
        if respuesta == diccionario[palabra]:
            print("¡Correcto!")
            aciertos += 1
    print(f"Puntuación final: {aciertos}")
