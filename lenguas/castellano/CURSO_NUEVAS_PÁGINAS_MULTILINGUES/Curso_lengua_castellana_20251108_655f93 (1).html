    map<string, string> palabras = {
        {"valla", "¿Valla o Vaya? [1]valla [2]vaya:2"},
        {"haber", "¿Haber o A ver? [1]haber [2]a ver:2"}
    };
    
    // Lógica del ejercicio
    for(const auto& [palabra, pregunta] : palabras) {
        cout << pregunta << endl;
        string respuesta;
        cin >> respuesta;
        // Verificar respuesta...
    }
    
    return 0;
}

fun main() {
    val palabras = mapOf(
        "valla" to "¿Valla o Vaya? [1]valla [2]vaya:2",
        "haber" to "¿Haber o A ver? [1]haber [2]a ver:2"
    )
    
    // Lógica del ejercicio
    palabras.forEach { (palabra, pregunta) ->
        println(pregunta)
        val respuesta = readLine()
        // Verificar respuesta...
    }
}
