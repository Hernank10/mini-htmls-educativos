Scaffolding: npm + Vite + Bootstrap (SASS)

    
Incluyo un scaffold mínimo para convertir esta demo en un proyecto con bundler, SASS y personalización de Bootstrap.

    
    

      
package.json

      
{
  "name": "centro-bilingue",
  "version": "1.0.0",
  "private": true,
  "scripts": {
    "dev": "vite",
    "build": "vite build",
    "preview": "vite preview"
  },
  "dependencies": {
    "bootstrap": "^5.3.2"
  },
  "devDependencies": {
    "vite": "^5.0.0",
    "sass": "^1.0.0"
  }
}

      
vite.config.js

      
import { defineConfig } from 'vite'
export default defineConfig({
  server: { port: 3000 }
})
