Bibliotecas para análisis (comentadas: instalar en entorno virtual) # pip install librosa soundfile matplotlib numpy import numpy as np import soundfile as sf import librosa import librosa.display import matplotlib.pyplot as plt UPLOAD_FOLDER = 'uploads' SPECTRO_FOLDER = 'static/spectrograms' ALLOWED_EXT = {'wav','flac','mp3','m4a'} app = Flask(__name__) app.config['UPLOAD_FOLDER'] = UPLOAD_FOLDER os.makedirs(UPLOAD_FOLDER, exist_ok=True) os.makedirs(SPECTRO_FOLDER, exist_ok=True) def allowed_file(filename): return '.' in filename and filename.rsplit('.',1)[1].lower() in ALLOWED_EXT @app.route('/') def index(): return render_template('index.html') @app.route('/voice') def voice(): return render_template('exercise_voice.html') @app.route('/api/analyze_voice', methods=['POST']) def analyze_voice(): if 'audioFile' not in request.files: return jsonify({'error':'No se subió ningún archivo'}), 400 f = request.files['audioFile'] if f.filename == '' or not allowed_file(f.filename): return jsonify({'error':'Archivo inválido o sin nombre'}), 400 filename = secure_filename(f.filename) unique = f"{uuid.uuid4().hex}_{filename}" path = os.path.join(app.config['UPLOAD_FOLDER'], unique) f.save(path) # Cargar audio try: y, sr = librosa.load(path, sr=None, mono=True) except Exception as e: return jsonify({'error':f'Error al leer audio: {e}'}), 500 duration = librosa.get_duration(y=y, sr=sr) # Calcular F0 con pyin try: f0, voiced_flag, voiced_probs = librosa.pyin(y, fmin=librosa.note_to_hz('C2'), fmax=librosa.note_to_hz('C7')) if np.any(voiced_flag): f0_mean = float(np.nanmean(f0[voiced_flag])) else: f0_mean = None except Exception: f0_mean = None # Intensidad aproximada (RMS -> dB) S, phase = librosa.magphase(librosa.stft(y)) rms = librosa.feature.rms(S=S) intensity_db = 20 * np.log10(np.maximum(rms, 1e-6)) intensity_mean = float(np.mean(intensity_db)) # Generar espectrograma (imagen) spec_name = f"spec_{uuid.uuid4().hex}.png" spec_path = os.path.join(SPECTRO_FOLDER, spec_name) plt.figure(figsize=(8,3)) D = np.abs(librosa.stft(y))**2 S_mel = librosa.feature.melspectrogram(S=D, sr=sr) S_db = librosa.power_to_db(S_mel, ref=np.max) librosa.display.specshow(S_db, sr=sr, x_axis='time', y_axis='mel') plt.title('Mel spectrogram') plt.tight_layout() plt.savefig(spec_path, bbox_inches='tight', dpi=150) plt.close() spec_url = url_for('static', filename=f'spectrograms/{spec_name}') return jsonify({ 'f0_mean': f0_mean, 'intensity_mean': intensity_mean, 'duration': duration, 'spectrogram_url': spec_url }) if __name__ == '__main__': app.run(debug=True) ``` --- ## Instrucciones rápidas 1. Crea un entorno virtual: `python -m venv venv && source venv/bin/activate` (Windows: `venv\Scripts\activate`). 2. Instala dependencias: `pip install flask librosa soundfile matplotlib numpy`. 3.
